cat .git/config مشاهده پیکربندی محلی
git --version نسخه ی git نصب شده را نمایش می دهد
git add -A بردن تمام فایل های تغییر کرده به staging index
git add [نام فایل] بردن فایل تعیین شده به staging index
git branch نمایش فهرست branchهای محلی و مشخص کردن شاخه فعال
git branch -r نمایش شاخه های remoteشده
git branch -a نمایش شاخه های محلی به همراه شاخه های remoteشده
git branch [نام شاخه جدید] ساخت branch جدید
git branch --move [current name] [new name]
git branch -m [current name] [new name]
 تغییر نام شاخه
git branch --delete [branch name]
git branch -d [branch name]
حذف شاخه
git checkout -- [نام فایل] دور انداختن تغییراتی که روی یک فایل خاص در working directory انجام شده و برگرداندن نسخه ی ذخیره شده ی آن در Repository Directory
git checkout [hash code] [نام فایل] مشاهده ی وضعیت یک فایل به خصوص در یک commit مشخص. تفاوت checkout با reset در این است که ما از checkout برای بررسی کارهای گذشته استفاده می کنیم اما از reset برای انتقال به یک نقطه ی خاص استفاده می شود.
git checkout [نام شاخه] جابه جایی بین شاخه ها - انتقال ه یک branch دیگر
git checkout -b [نام شاخه] شاخه جدید می سازد و بلافاصله به آن checkout می کند
git clean -n این دستور فایل هایی که track نمی شن رو نمایش میده.
git clean -f فایل هایی که track نمی شن پاک می شوند.
git clean -i علاوه بر نمایش فایل های track نشده یک منو نمایش داده می شود که می توان از آن برای حذف آن فایل ها استفاده کرد.
git commit -m [پیغام] commit کردن تغییرات از staging index به Repository Directory
git commit -am [پیغام] این دستور فایلهای ویرایش شده و حذف شده را به staging index می فرستد و سپس commit میکند. فایل های جدیدی که track نمی شوند را اضافه نمی کند.
git commit --amend -m [پیغام] این دستور تغییرات جدید را جایگزین آخرین commit می کند و پیغام جدید را ثبت می کند
git config --list نمایش تمام تنظیمات پیکربندی چه محلی و چه سراسری
git config --local --list نمایش فهرست تنظیمات پیکربندی محلی
git config --global --list نمایش فهرست تنظیمات پیکربندی سراسری
git config --global user.name [نام کاربر] تغییر نام کاربر در فایل پیکربندی سراسری
git config user.name [نام کاربر] تغییر نام کاربر در فایل پیکربندی محلی
git config --global user.email [ایمیل کاربر] تغییر ایمیل کاربر در فایل پیکربندی سراسری
git config user.email [ایمیل کاربر] تغییر ایمیل کاربر در فایل پیکربندی محلی
git config --global core.excludefile [مسیر و نام فایل] استفاده از فایل .gitignore سراسری به جای فایل .gitignore محلی
git config --global alias.[shortcut] "[command without git]" ساخت دستور میانبر
git diff نمایش تفاوتهای فایل های تغییر کرده در working directory با همتاهای آنها در staging index
git diff --staged نمایش تفاوتهای فایل های staging index با همتاهای آنها در repository directory
git diff --word-diff
git diff --color-words		 نمایش تفاوتها به صورت کلمه به کلمه
git diff [نام شاخه]..[نام شاخه] تفاوت فایل های مختلف دو شاخه ی مختلف را نمایش می دهد
git [دستور] --help
git help [دستور]
git --help [دستور]		 نمایش توضیحات دستور
git init شروع کردن کار گیت برای یک پوشه
git log نمایش فهرست کامیت های انجام شده
git log --oneline نمایش فهرست عنوان تمام commitها بدون توضیحات
git log --format=oneline نمایش فهرست عنوان تمام commitها بدون توضیحات به همراه کد کامل SHA-1
git log --until="[2019-03-30 12:00:00]" commitهایی که قبل از تاریخ تعیین شده انجام شدند را فهرست می کند
git log --after="[2019-04-01 20:00:00]" commitهایی که بعد از تاریخ تعیین شده انجام شدند را فهرست می کند
git log --until="4 days ago" commitهایی که قبل از چهار روز پیش انجام شدند را نمایش می دهد
git log --after="2 weeks ago" commitهایی که بعد از دو هفته پیش انجام شدند را نمایش می دهد
git log --until="yesterday" commitهایی که تا دیروز انجام شدند را نمایش می دهد
git log --author="[Morteza]" commitهایی که توسط Morteza انجام شدند را نمایش می دهد
git log -[3] نمایش فهرست سه commit آخر
git log --grep="[background]" commitهایی که در توضیحاتشون کلمه ی background وجود داشته باشد را نمایش می دهد
git log [a4b57a1].. commitهایی که بعد از a4b57a1 انجام شدند را نمایش می دهد
git log [index.html] تمام commitهایی که برای index.html انجام شده است را نمایش می دهد
git log -p commitها را به همراه توضیحاتی در مورد تغییرات و تفاوت هایشان نمایش می دهد
git log --stat تعداد خطوط حذف و اضافه شده در هر commit را نمایش می دهد
git log --summary توضیحات مختصری در رابطه با commit نشان داده می شود
git ls-tree [HEAD/master] نمایش فهرست فایل ها و پوشه ها به همراه کد SHA-1 مربوط به هر کدام
git merge [نام شاخه] شاخه ی master را با شاخه ی تعیین شده merge می کند. تمام commitها را حفظ میکند. تمام commitهای شاخه ی master و شاخه ی تعیین شده را نگه میدارد و در یک commit جدید عملیات ادغام را انجام می دهد.
git merge --squash [نام شاخه] تمام commitهای شاخه ی تعیین شده را با هم merge می کند و یکجا با آخرین commit شاخه ی اصلی merge می کند.
git merge --abort صرف نظر کردن از merge در صورتی که conlict رخ داده باشد
git mv [نام فعلی] [نام جدید] نام فایل را تغییر می دهد و این تغییر را در staging index قرار می دهد
git mv [مسیر فعلی] [مسیر جدید] تغییر مسیر فایل
git push -u [alias] [branch] ارسال پروژه موجود برای remote
git rebase [نام شاخه] rebase بر خلاف merge ساده، برای ادغام کردن یک commit جدید نمی سازد و ادغام را با آخرین commit شاخه master انجام می دهد.
git remote add [alias(origin)] [url] اضافه کردن remote
git remote مشاهده نام remote
git remote -v مشاهده urlهای fetch و push
git remote rm [alias] حذف remote
git reset HEAD بیرون کشیدن فایلها از staging index
git reset HEAD [نام فایل] بیرون کشیدن یک فایل مشخص از staging index
git reset --soft [hash code] محتوای repository directory تعیین شده را به staging index منتقل میکند. repository را به مرحله ی قبل بر می گرداند.
git reset --mixed [hash code] repository را به مرحله ی قبل بر می گرداند. اگر پارامتری تعیین نشود mixed در نظر گرفته می شود.
git reset --hard [hash code] محتوای یک repository directory خاص را به working directory منتقل می کند. در واقع بازگشت به یک نقطه ی خاص است.
git rm [نام فایل] حذف فایل و قرار دادن این تغییر در staging index
git rm --cached [نام فایل] حذف فایل از repository directory بدون آنکه از working directory حذف شود
git show [hash-code] فاوت ها و تغییرات commit مشخص شده را نمایش می دهد. میشه از کدهای HASH مربوط به یک پوشه یا فایل در خروجی دستور ls-tree برای نمایش فایلها و پوشه های درون یک پوشه یا نمایش محتوای یک فایل استفاده کرد
git show HEAD^^/HEAD~2 تفاوت ها و تغییرات مربوط به دو commit قبل از HEAD (پدر پدر HEAD) را نمایش می دهد
git status نمایش وضعیت تغییرات
git stash save زمانی که در یک شاخه تغییرات commit نشده داریم نمی توانیم به شاخه ی دیگر checkout کنیم. یکی از راه حل ها این است که از این دستور استفاده کنیم و تغییرات را در انبار ذخیره کنیم.
git stash list فهرست stashها یا موارد ذخیره شده در انبار را نمایش می دهد.
git stash show stash@{0} نمایش آمار تغییرات مربوط به انبار stash 0
git stash show -p stash@{0} نمایش تفاوت های مربوط به انبار stash 0
git stash pop stash@{1} stash انتخاب شده را به working directory بر می گرداند و آن را از فهرست انبار stash حذف می کند. اگه مشخص نشود کدام stash، از آخر فهرست stash ها بر می دارد.
git stash apply اعمال تغییرات یک مورد از انبار stach روی working directory بدون اینکه آن مورد از انبار stash حذف شود
git stash clear پاک کردن انبار stash
git stash drop stash@{2} حذف یک مورد از انبار stash
